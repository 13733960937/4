plane_bullet (英雄飞机子弹模块)

# 导入模块
import pygame

# 定义两个常量 (一经赋值 不能改变) 记录窗口的高度和宽度
WINDOW_HEIGHT = 768
WINDOW_WIDTH = 512

# 自定义子弹类
class PlaneBullet(object):

    # 构造函数
    def __init__(self):
        # 加载资源图片
        self.img = pygame.image.load("res/bullet_10.png")
        # 图片矩形
        self.img_rect = self.img.get_rect()
        # 速度
        self.speed = 4
        # 是否发射 -> 未发射
        self.is_shot = False

    # 向上移动
    def move_up(self):
        # 向上移动
        self.img_rect.move_ip(0, -self.speed)

        # 超出了屏幕就以为已经死亡 没有发射
        if self.img_rect[1] <= -self.img_rect[3]:
            # 超出屏幕设置未发射
            self.is_shot = False
