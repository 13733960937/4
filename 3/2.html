game_map (飞机大战地图模块)

# 导入模块
import pygame, random

# 定义两个常量 (一经赋值 不能改变) 记录窗口的高度和宽度
WINDOW_HEIGHT = 768
WINDOW_WIDTH = 512

# 自定义一个地图类-> 管理地图图片
class GameMap(object):

    # 构造函数
    def __init__(self):
        #随机选择地图
        self.num = str(random.randint(1, 5))
        # 地图图片两张图片
        self.bg_img1 = pygame.image.load("res/img_bg_level_" + self.num + ".jpg")
        self.bg_img2 = pygame.image.load("res/img_bg_level_" + self.num + ".jpg")

        # 定义当前两张图片的y轴坐标
        self.img1_y = - WINDOW_HEIGHT
        self.img2_y = 0

        # 定义地图的移动速度
        self.speed = 2

    # 叫地图图片动起来
    def move(self):
        # 判断下临界值
        if self.img1_y >= 0:
            # y轴重置
            self.img1_y = - WINDOW_HEIGHT
            self.img2_y = 0

        # 做动画
        self.img1_y += self.speed
        self.img2_y += self.speed
