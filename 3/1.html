game_main (飞机大战主模块)

# 导入模块
import pygame, sys, random, game_map, hero_plane, enemy_plane, game_score

# 定义两个常量 (一经赋值 不能改变) 记录窗口的高度和宽度
WINDOW_HEIGHT = 768
WINDOW_WIDTH = 512

# 自定义一个游戏窗口类
class GameWindow(object):

    # 类方法 -> 记录分数
    __cls_score = 0

    # 构造函数 -> 关闭窗口
    def __init__(self):
        # 实例化pygame
        pygame.init()
        # 播放背景音乐
        self.set_game_music()
        # 加载音效
        self.boom_sound = pygame.mixer.Sound("./res/baozha.ogg")      
        # 实例化一个游戏窗口
        self.window = pygame.display.set_mode([WINDOW_WIDTH, WINDOW_HEIGHT])
        # 设置窗口的title(名字)
        pygame.display.set_caption("飞机大战")
        # 加载资源文件->  图片
        logo_image = pygame.image.load("res/app.ico")
        # 设置窗口的logo
        pygame.display.set_icon(logo_image)
        # 实例化一个地图对象
        self.map = game_map.GameMap()
        # 实例化英雄飞机对象
        self.hero_plane = hero_plane.HeroPlane()
        # 实例化敌机机群
        self.enemy_plane_list = [enemy_plane.EnemyPlane() for _ in range(5)]
        # 文字对象
        self.game_score = game_score.GameScore(30)

    # 播放背景音乐
    def set_game_music(self):
        # 加载背景音乐
        pygame.mixer.music.load("./res/bg2.ogg")
        # 循环播放背景音乐 -1 无限循环
        pygame.mixer.music.play(-1)

    # 01-处理矩形移动
    def __action(self):
        # 执行地图动画 -> map
        self.map.move()

        # 让子弹飞一会
        for bullet in self.hero_plane.bullet_list:
            if bullet.is_shot:
                bullet.move_up()
        # 让敌机动起来
        for enmey in self.enemy_plane_list:
            enmey.move_down()


    # 02-绘制元素
    def __draw(self):
        # 添加地图类中两张图片
        self.window.blit(self.map.bg_img1, (0, self.map.img1_y))
        self.window.blit(self.map.bg_img2, (0, self.map.img2_y))

        # 添加英雄飞机
        self.window.blit(self.hero_plane.img, (self.hero_plane.img_rect[0], self.hero_plane.img_rect[1]))

        # 添加子弹 -> 指绘制已经发射的
        for bullet in self.hero_plane.bullet_list:
            if bullet.is_shot:
                self.window.blit(bullet.img, (bullet.img_rect[0], bullet.img_rect[1]))

        # 添加敌机
        for enemy in self.enemy_plane_list:
            self.window.blit(enemy.img, (enemy.img_rect[0], enemy.img_rect[1]))

        # 文字
        self.window.blit(self.game_score.text_obj, (0, 0))

    # 03-处理窗口的事件监听
    def __event(self):
        # 获取事件列表
        event_list = pygame.event.get()
        # 遍历事件列表
        for event in event_list:
            # 鼠标点击
            if event.type == pygame.QUIT:
                    self.__game_over()

            # 按钮点击
            if event.type == pygame.KEYDOWN:
                # esc
                if event.key == pygame.K_ESCAPE:
                    self.__game_over()
                # 空格
                if event.key == pygame.K_SPACE:
                    self.hero_plane.shot()

        # 获取所有按钮元组
        keys_pressed = pygame.key.get_pressed()
        # 长按up
        if keys_pressed[pygame.K_UP]:
            self.hero_plane.move_up()
        # 长按down
        if keys_pressed[pygame.K_DOWN]:
            self.hero_plane.move_down()
        # 长按left
        if keys_pressed[pygame.K_LEFT]:
            self.hero_plane.move_left()
        # 长按right
        if keys_pressed[pygame.K_RIGHT]:
            self.hero_plane.move_right()

    # 04-刷新窗口
    def __update(self):
        # 刷新
        pygame.display.update()

    # 子弹击中敌机
    def __bullet_hit_enmeyplane(self):

        # 遍历子弹
        for bullet in self.hero_plane.bullet_list:
            # 如果子弹是发射状态
            if bullet.is_shot:

                # 在遍历敌机
                for enemy in self.enemy_plane_list:

                    # 如果击中
                    if pygame.Rect.colliderect(bullet.img_rect, enemy.img_rect):
                        # 子弹设置为死亡
                        bullet.is_shot = False
                        # 飞机要重置
                        enemy.reset()
                        # 分数累加
                        GameWindow.__cls_score += 5
                        # 设置文字
                        self.game_score.set_text_obj(("分数:%d" %GameWindow.__cls_score))
                        # 设置音效
                        self.__set_boom_sound()

    # 播放爆炸音效
    def __set_boom_sound(self):
        # 播放音效
        self.boom_sound.play()

    # 游戏结束
    def __game_over(self):
        print("游戏结束")
        # 停止背景音乐
        pygame.mixer.music.stop()
        # 停止音效
        self.boom_sound.stop()

        pygame.quit()
        sys.exit()


    # 执行程序 -> 死循环
    def run(self):
        # 死循环
        while True:
            self.__action()
            self.__draw()
            self.__event()
            self.__update()
            self.__bullet_hit_enmeyplane()


# 程序的入口函数
def main():
    # 实例化一个对象
    game = GameWindow()
    # 执行run
    game.run()


if __name__ == '__main__':
    # 执行主函数
    main()
