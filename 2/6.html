英雄飞机装备子弹并发射

# 导入模块
import pygame
import random
import sys

# 定义常量 记录窗口的宽和高
WINDOW_WIDTH = 512
WINDOW_HEIGHT = 768

# 自定义子弹类
class PlaneBullet():
    # 构造函数
    def __init__(self):
        # 加载子弹图片
        self.img = pygame.image.load("res/bullet_14.png")
        # 获取子弹的矩形
        self.img_rect = self.img.get_rect()
        # 子弹速度
        self.speed = 4
        # 是否发射
        self.is_shot = False

    # 子弹向上移动
    def move_up(self):
        # 开始移动
        self.img_rect.move_ip(0, -self.speed)
        # 判断如果子弹超出窗口 设置为False
        if self.img_rect[1] <= - self.img_rect[3]:
            self.is_shot = False


# 自定义英雄飞机类
class HeroPlane(object):
    # 构造函数
    def __init__(self):
        # 加载飞机图片
        self.img = pygame.image.load("res/hero2.png")
        # 获取飞机图片矩形
        self.img_rect = self.img.get_rect()
        # 在原基础基础上 移动指定的偏移量(x, y)
        self.img_rect.move_ip((WINDOW_WIDTH - self.img_rect[2])/2, WINDOW_HEIGHT - self.img_rect[3] - 30)
        # 飞机的速度
        self.speed = 2
        # 飞机装载弹夹 子弹是飞机的一部分
        self.bullet_list = [PlaneBullet() for _ in range(10)]

    # 向上移动
    def move_up(self):
        if self.img_rect[1] > 0:
            self.img_rect.move_ip(0, -self.speed)

    # 向下移动
    def move_down(self):
        if self.img_rect[1] < WINDOW_HEIGHT - self.img_rect[3]:
            self.img_rect.move_ip(0, self.speed)
    # 向左移动
    def move_left(self):
        if self.img_rect[0] > 0:
            self.img_rect.move_ip(-self.speed, 0)

    # 向右移动
    def move_right(self):
        if self.img_rect[0] < WINDOW_WIDTH - self.img_rect[2]:
            self.img_rect.move_ip(self.speed, 0)

    # 发射子弹
    def shot(self):
        # 遍历弹夹列表
        for bullet in self.bullet_list:
            # 判断子弹是未发射状态
            if not bullet.is_shot:
                # 设置子弹的开始位置坐标(x, y)
                bullet.img_rect[0] = self.img_rect[0] + (self.img_rect[2] - bullet.img_rect[2])/2
                bullet.img_rect[1] = self.img_rect[1] - bullet.img_rect[3]
                # 子弹设置为已发射状态
                bullet.is_shot = True
                # 每次只能发射一颗
                break


# 自定义地图类
class GameMap(object):
    # 构造函数
    def __init__(self):
        # 设置随机数
        self.num = str(random.randint(1, 5))
        # 加载图片资源
        self.bg_img1 = pygame.image.load("res/img_bg_level_" + self.num + ".jpg")
        self.bg_img2 = pygame.image.load("res/img_bg_level_" + self.num + ".jpg")

        # 记录两张图片的y轴坐标
        self.img1_y = -WINDOW_HEIGHT
        self.img2_y = 0

        # 地图移动速度
        self.speed = 2

    # 地图向下移动
    def move(self):
        # 如果已经超出屏幕后 应该重置
        if self.img1_y >= 0:
            self.img1_y = -WINDOW_HEIGHT
            self.img2_y = 0

        # 执行地图动画
        self.img1_y += self.speed
        self.img2_y += self.speed


# 自定义游戏窗口类
class GameWindow(object):
    # 构造函数
    def __init__(self):
        # 初始化pygame
        pygame.init()
        # 实例化窗口对象
        self.window = pygame.display.set_mode([WINDOW_WIDTH, WINDOW_HEIGHT])
        # 加载资源图片
        self.icon_img = pygame.image.load("res/app.ico")
        # 设置窗口icon
        pygame.display.set_icon(self.icon_img)
        # 设置窗口标题
        pygame.display.set_caption("飞机大战")

        # 实例化地图对象
        self.map = GameMap()
        # 实例化英雄飞机对象
        self.hero_plane = HeroPlane()

    # 1.处理各种矩形坐标移动
    def action(self):
        # 执行地图动画
        self.map.move()
        # 指定子弹动画
        for bullet in self.hero_plane.bullet_list:
            if bullet.is_shot:
                bullet.move_up()

    # 2.根据矩形坐标，重新对元素进行绘制
    def draw(self):
        # 添加地图图片
        self.window.blit(self.map.bg_img1, (0, self.map.img1_y))
        self.window.blit(self.map.bg_img2, (0, self.map.img2_y))
        # 添加英雄飞机
        self.window.blit(self.hero_plane.img, (self.hero_plane.img_rect[0], self.hero_plane.img_rect[1]))
        # 添加弹夹 只绘制子弹状态是以发射的
        for bullet in self.hero_plane.bullet_list:
            if bullet.is_shot:
                self.window.blit(bullet.img, (bullet.img_rect[0], bullet.img_rect[1]))

    # 3.处理窗口中的监听事件
    def event(self):
        # 获取所有监听事件列表
        event_list = pygame.event.get()
        # 遍历列表
        for event in event_list:
            # 监听鼠标退出 -> 退出游戏
            if event.type == pygame.QUIT:
                self.game_over()
            # 监听按钮点击
            if event.type == pygame.KEYDOWN:
                # 如果是esc键 -> 退出游戏
                if event.key == pygame.K_ESCAPE:
                    self.game_over()

                # 如果是空格键 -> 发射子弹
                if event.key == pygame.K_SPACE:
                    self.hero_plane.shot()

        # 监听上下左右按钮点击
        pressed = pygame.key.get_pressed()
        # up
        if pressed[pygame.K_UP]:
            self.hero_plane.move_up()
        # down
        if pressed[pygame.K_DOWN]:
            self.hero_plane.move_down()
        # left
        if pressed[pygame.K_LEFT]:
            self.hero_plane.move_left()
        # right
        if pressed[pygame.K_RIGHT]:
            self.hero_plane.move_right()

    # 4.刷新窗口
    def update(self):
        pygame.display.update()

    # 游戏结束
    def game_over(self):
        # 停止pygame引擎
        pygame.quit()
        # 结束程序
        sys.exit()


    # 开始执行程序
    def run(self):
        # 死循环
        while True:
            # 处理矩形移动
            self.action()
            # 绘制元素
            self.draw()
            # 处理窗口的事件监听
            self.event()
            # 刷新窗口
            self.update()


# 程序的入口
def main():
    # 实例化game
    game = GameWindow()
    # 执行程序
    game.run()
    print("哈哈")


if __name__ == '__main__':
    main()
